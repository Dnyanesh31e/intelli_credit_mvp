elif st.session_state.stage == 2:
    st.markdown('<div class="stage-header"><h2>Stage 2: Intelligent Data Ingestion</h2></div>', unsafe_allow_html=True)

    entity = None
    if st.session_state.entity_id:
        entity = entity_manager.get_entity(st.session_state.entity_id)

    if entity:
        st.markdown(f"**Entity:** {entity['basic_info']['company_name']} | **ID:** {st.session_state.entity_id}")

        st.markdown("**Supported formats:** PDF, DOCX, XLSX, XLS, TXT, PNG, JPG, JPEG")
        st.markdown("**Tip:** You can upload multiple files at once or one by one!")

        # Multi-file uploader

            "Upload one or more documents (select multiple files)",
            type=["pdf", "docx", "xlsx", "xls", "txt", "png", "jpg", "jpeg"],
            accept_multiple_files=True,
            key="multi_file_uploader"
        )

        # Document type mapping
        doc_types = {
            "alm": "📊 ALM (Asset-Liability Management)",
            "shareholding": "👥 Shareholding Pattern",
            "borrowing": "💰 Borrowing Profile",
            "annual_report": "📈 Annual Reports (P&L, Balance Sheet, Cash Flow)",
            "portfolio": "📁 Portfolio Cuts/Performance Data"
        }

        # Process uploaded files
        if uploaded_files:
            st.markdown("### Uploaded Files")
            st.markdown(f"**Total files uploaded:** {len(uploaded_files)}")

            for idx, uploaded_file in enumerate(uploaded_files):
                file_name = uploaded_file.name
                file_size = uploaded_file.size
                file_type = uploaded_file.type

                # Determine document type from filename or let user select
                col1, col2, col3 = st.columns([3, 2, 1])

                with col1:
                    st.markdown(f"**📄 {file_name}**")
                    st.caption(f"Size: {file_size / 1024:.1f} KB | Type: {file_type}")

                with col2:
                    # Auto-detect doc type from filename
                    detected_type = None
                    file_lower = file_name.lower()
                    if "alm" in file_lower:
                        detected_type = "alm"
                    elif "share" in file_lower or "holding" in file_lower:
                        detected_type = "shareholding"
                    elif "borrow" in file_lower or "debt" in file_lower:
                        detected_type = "borrowing"
                    elif "annual" in file_lower or "financial" in file_lower or "report" in file_lower:
                        detected_type = "annual_report"
                    elif "portfolio" in file_lower or "performance" in file_lower:
                        detected_type = "portfolio"

                    default_index = list(doc_types.keys()).index(detected_type) if detected_type else 0

                    selected_type = st.selectbox(
                        "Document Type",
                        list(doc_types.keys()),
                        format_func=lambda x: doc_types[x],
                        index=default_index,
                        key=f"doc_type_{idx}"
                    )

                with col3:
                    if st.button("🗑️", key=f"remove_{idx}", help="Remove file"):
                        pass  # File will be removed on next rerun

                st.divider()

                # Store in session state
                if selected_type not in st.session_state.uploaded_docs:
                    st.session_state.uploaded_docs[selected_type] = {
                        "file_name": file_name,
                        "file_size": file_size,
                        "file_type": file_type,
                        "uploaded_at": datetime.now().isoformat(),
                        "file_data": uploaded_file.getvalue()
                    }

        # Show upload progress
        st.markdown("### Upload Progress")
        uploaded_types = list(st.session_state.uploaded_docs.keys())
        missing_types = [k for k in doc_types.keys() if k not in uploaded_types]

        col1, col2 = st.columns(2)

        with col1:
            st.markdown("**Uploaded:**")
            for doc_type in uploaded_types:
                st.markdown(f"✅ {doc_types[doc_type]}")

        with col2:
            st.markdown("**Missing:**")
            for doc_type in missing_types:
                st.markdown(f"⬜ {doc_types[doc_type]}")

        st.progress(len(st.session_state.uploaded_docs) / 5)

        st.markdown(f"**Progress:** {len(st.session_state.uploaded_docs)}/5 documents uploaded")

            if st.button("✅ Proceed to Extraction & Classification", type="primary", use_container_width=True):

        else:

                    st.session_state.stage = 3
